How to create local SQL DB:

1.Open command prompt as administrator

2. type sqllocaldb create "Local"

3. SQL DB will be created

 

How to connect to local DB

1. Give Server name as "(localdb)\Local"

2. Choose Windows Authentication from the dropdown of Authentication mode

3. Click connect

 



SQL SCRIPTS

----------------------

1. create database MedicaidFAQ

 

Tables

1. 

create table Modules

(

  MODULE_ID  int  not null,

  MODULE_NAME  varchar(2000)  not null

)

 

2. 

create table USERROLE

(

  USERROLE_ID  int  primary key,

  USERROLE_NAME  varchar(2000)  not null

)

 

3.

create table ENVIRONMENT

(

  ENVIRONMENT_ID  int  primary key,

  ENVIRONMENT_NAME  varchar(2000)  not null

)

 

4.

create table PROBLEMTYPE

(

  PROBLEMTYPE_ID  int  primary key,

  PROBLEMTYPE_NAME  varchar(2000)  not null

)

 

5.

create table SOLUTION

(

  SOLUTION_ID  int identity(100,1)  primary key,

  MODULE_ID  int  not null,

  USERROLE_ID  int  not null,

  ENVIRONMENT_ID  int  null,

  PROBLEMTYPE_ID  int  null,

  SOLUTION_DESC  nvarchar(max)

)

5.

CREATE PROCEDURE InsertSolutionInformation_SP

@MODULE_ID int,

@USERROLE_ID int,

@ENVIRONMENT_ID int,

@PROBLEMTYPE_ID int,

@SOLUTION_DESC nvarchar(max)

AS BEGIN

   INSERT into SOLUTION(MODULE_ID,USERROLE_ID,ENVIRONMENT_ID,PROBLEMTYPE_ID,SOLUTION_DESC)

   VALUES(@MODULE_ID,@USERROLE_ID,@ENVIRONMENT_ID,@PROBLEMTYPE_ID,@SOLUTION_DESC)

END
